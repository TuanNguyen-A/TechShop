/*1)ROLE*/ 
create table role(
    id int primary key AUTO_INCREMENT,
    name varchar(20)
)

/*2)USER*/ 
create table users(
    id int primary key AUTO_INCREMENT,
    fullname varchar(50),
    email varchar(150),
    address varchar(200),
    password varchar(50),
    phone_number varchar(12),
    role_id int references role(id),
    created_at datetime,
    updated_at datetime,
    deleted int
)

/*3)LOGIN TOKEN */
create table login_token(
    id_user int references users(id), 
    token varchar(50) not null unique,
    primary key (id_user, token)
)

/*4)CATEGORY*/
create table category(
    id int primary key AUTO_INCREMENT,
    name varchar(100)
)

/*5)PRODUCT */
create table products(
    id int primary key AUTO_INCREMENT,
    category_id int references category(id),
    title varchar(350),
    price int,
    discount int,
    thumbnail varchar(500),
    description longtext,
    created_at datetime,
    updated_at datetime,
    deleted int
)

/*6)GALERY*/
create table galery(
    id int primary key AUTO_INCREMENT,
    product_id int references products(id),
    thumbnail varchar(500)
)

/*7)FEEDBACK*/
create table feedback(
    id int primary key AUTO_INCREMENT,
    firstname varchar(30),
    lastname varchar(30),
    email varchar(150),
    phone_number varchar(20),
    subject_name varchar(200),
    note varchar(500)
)

/*8)ORDER */
create table orders(
    id int primary key AUTO_INCREMENT,
    user_id int references users(id),
    fullname varchar(50),
    email varchar(150),
    address varchar(200),
    phone_number varchar(12),
    note varchar(500),
    order_date datetime,
    status int,
    total_money int
)

/*9)ORDER_DETAILS*/
create table order_details(
    id int primary key AUTO_INCREMENT,
    order_id int references orders(id),
    product_id int references products(id),
    price int,
    num int,
    total_money int
)
